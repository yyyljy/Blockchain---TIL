## useEffect

- 타이머 간격 변경 연습

```javascript
import { useState, useEffect } from "react";
function Page() {
  const [count, setCount] = useState(0);
  const [delay, setDelay] = useState(1000);
  useEffect(() => {
    const id = setInterval(() => {
      setCount((count) => count + delay);
    }, delay * 1000);
    return () => {
      clearInterval(id);
    };
  }, [delay]);
  const seconds = [0.5, 1, 3, 5];
  return (
    <>
      <p>안녕하세요</p>
      <p>이 페이지에 들어오신지</p>
      <div>
        <h3>타이머의 간격이 바뀌도록 하세요</h3>
        {seconds.map((s) => (
          <button
            onClick={() => {
              setDelay(s);
            }}
          >
            {s}
          </button>
        ))}
      </div>
      <p>
        <strong>{count}</strong>초 되었습니다.
      </p>
    </>
  );
}

export default Page;
```

- 의존성배열에 넣어준 값이 바뀔 때마다 useEffect가 실행됨
  - useEffect 맨 마지막 배열 요소



## useReducer

- reducer state가 어떤 action이 취해졌을 때 어떻게 변한지 시나리오 작성 가능
- reducer 함수는 state, action

```javascript
import { useReducer } from "react";

function Counter() {
  //시나리오
  //+1 버튼 누르면 state가 +1 된다
  //-1 버튼 누르면 state가 -1 된다
  // action => {type:"action의 이름(숫자증가", payload:인자}
  function reducer(state, action) {
    switch (action.type) {
      case "INCREMENT":
        return state + 1;
      case "DECREMENT":
        return state - 1;
      default:
        throw new Error("Not Defined Error.");
    }
  }
  const [number, dispatch] = useReducer(reducer, 0);
  return (
    <>
      <h1>숫자:{number}</h1>
      <div>
        <button onClick={() => dispatch({ type: "INCREMENT" })}>+1</button>
        <button onClick={() => dispatch({ type: "DECREMENT" })}>-1</button>
      </div>
    </>
  );
}

export default Counter;
```



## useRef

- useState는 불변성을 이용함
- useRef는 의도적으로 참조하여 변경 가능하도록 함

```javascript
import { useRef, useState } from "react";

function Timer() {
  const [time, setTime] = useState(0);
  const [isRunning, setIsRunnning] = useState(false);
  const timerRef = useRef(null);
  function handleStart() {
    timerRef.current = setInterval(() => {
      setTime((t) => t + 1);
    }, 1000);
    setIsRunnning(true);
  }

  function handleStop() {
    if (timerRef.current) {
      clearInterval(timerRef.current);
      setIsRunnning(false);
      timerRef.current = null;
    }
  }

  return (
    <>
      <h1>타이머: {time}</h1>
      <button onClick={handleStart}>시작</button>
      <button onClick={handleStop}>멈춤</button>
    </>
  );
}

export default Timer;
```

- 랜더링 할 때마다 id값이 바뀌어 clear가 안되는 현상을 ref로 해결



### Focus

import { useRef } from "react";

  const inputRef = useRef(null);
